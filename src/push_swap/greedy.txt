#include "libft.h"
#include "push_swap.h"
#include "ft_printf.h"

void	push_b(t_list **a, t_list **b, t_stack **a_stk, t_stack **b_stk)
{
	int	cnt;
	int	size;
	int	range;
	int	max_rotate;

	cnt = 5;
	size = ft_lstsize(*a);
	range = size / cnt;
	while (cnt)
	{
		max_rotate = 0;
		while (ft_lstsize(*b) < size - 2)
		{
			if ((*a)->idx <= range)
			{
				move_push(b, a, b_stk, a_stk);
				ft_printf("pb\n");
			}
			else
			{
				move_rotate(a, a_stk);
				ft_printf("ra\n");
			}
			max_rotate++;
			if (size == max_rotate || (*a_stk)->min > range)
				break ;
		}
		range += size / 5 + 1;
		cnt--;
	}
}

void	passing_to_b(t_list *a, t_list *b)
{
	t_stack	*a_stk;
	t_stack	*b_stk;

	a_stk = ft_new_stack(a);
	b_stk = ft_new_stack(b);
	push_b(&a, &b, &a_stk, &b_stk);
	sort(&a, &b, &a_stk, &b_stk);
}
